Sloppy 0.1.0-alpha.3
Pre-alpha runtime foundation with bounded run and metadata CLI introspection.

Usage:
  sloppy --help
  sloppy --version
  sloppy create <name> [--template api|minimal-api|program|cli|package-api|node-compat] [--force]
                       [--no-git] [--format text|json]
  sloppy build [source.js|source.mjs|source.ts] [--out <dir>] [--kind web|program]
               [--environment Development] [--host 127.0.0.1] [--port <port>]
  sloppy package [source.js|source.mjs|source.ts] [--out <dir>] [--kind web|program]
                 [--environment Development] [--host 127.0.0.1] [--port <port>]
                 [--format text|json]
  sloppy dev [source.js|source.mjs|source.ts] [--host 127.0.0.1] [--port <port>]
             [--watch|--no-watch] [--clear] [--openapi] [--env Development]
  sloppy run [source.js|source.mjs|source.ts|artifacts-dir|package-dir|--artifacts <dir>] [--stdlib <dir>]
             [--kind web|program] [--environment Development] [--host 127.0.0.1] [--port <port>]
             [--once METHOD TARGET] [--header 'Name: value'] [--body text]
             [--body-file path] [--json json] [--diagnostics-json] [-- <program-args...>]
  sloppy routes [artifacts-dir|plan.json|--plan <path>|--artifacts <dir>] [--format text|json] [--dispatch]
  sloppy health [artifacts-dir|plan.json|--plan <path>|--artifacts <dir>] [--format text|json]
  sloppy metrics [artifacts-dir|plan.json|--plan <path>|--artifacts <dir>] [--format text|json]
  sloppy capabilities [artifacts-dir|plan.json|--plan <path>|--artifacts <dir>] [--format text|json]
  sloppy deps [artifacts-dir|plan.json|--plan <path>|--artifacts <dir>] [--format text|json] [--explain]
  sloppy doctor [artifacts-dir|plan.json|--plan <path>|--artifacts <dir>] [--format text|json] [--dispatch] [--report]
  sloppy db status|migrate [artifacts-dir|package-dir] --provider <name> [--format text|json]
  sloppy orm migration add <name> [artifacts-dir|package-dir] --provider <name>
  sloppy orm migration script|status|apply [artifacts-dir|package-dir] --provider <name> [--format text|json]
  sloppy audit [artifacts-dir|plan.json|--plan <path>|--artifacts <dir>] [--format text|json]
  sloppy openapi [artifacts-dir|plan.json|--plan <path>|--artifacts <dir>] [--output <path>] [--strict|--fail-on-partial] [--summary] [--pretty]
